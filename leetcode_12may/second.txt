class Solution {
public:
    int maximumEnergy(vector<int>& e, int k) {
        int sm,mx=INT_MIN;;
        for(int i=e.size()-1;i>e.size()-1-k;i--){
            sm=0;
            for(int j=i;j>=0;j-=k){
                sm+=e[j];
                mx = max(mx,sm);
            }
        }
        return mx;
    }
};
