class Solution {
public:
    int numberOfSpecialChars(string word) {
        set<int>s1,s2;
        for(int i=0;i<word.size();i++){
            if(word[i]>='a' && word[i]<='z')
            s1.insert(word[i]-'a');
            else
            s2.insert(word[i]-'A');
        }
    int c=0;
        for(auto i:s1){
            if(s2.find(i)!=s2.end()) c++;
        }
        return c;
    }
};
